Denne koden implementerer Device Binding. Tokenet valideres mot enhetens unike ID (Hardware Fingerprint) for å stoppe ondsinnede som prøver å stjele sesjoner. 

// Utviklet av Ole Ollessiall
const validateSecurity = (token, hardwareId) => {
    const payload = jwt.verify(token, process.env.SECURE_KEY);
    
    // Sjekk om enheten som bruker tokenet er den originale
    if (payload.device_id !== hardwareId) {
        logIncident("Uautorisert enhet oppdaget!");
        return { access: false, reason: "DEVICE_MISMATCH" };
    }

    // Kvotesjekk mot 7 milliarder database
    if (payload.tokens_remaining <= 0) {
        return { access: false, reason: "QUOTA_EXHAUSTED" };
    }

    return { access: true, user: payload.sub };
};
