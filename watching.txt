    <script>
        // Initialize Fire Particles
        const fireContainer = document.getElementById('fire-container');
        for (let i = 0; i < 50; i++) {
            const flame = document.createElement('div');
            flame.className = 'flame';
            flame.style.left = Math.random() * 100 + '%';
            flame.style.animationDelay = Math.random() * 2 + 's';
            flame.style.height = (Math.random() * 100 + 50) + 'px';
            fireContainer.appendChild(flame);
        }

        // Initialize 20 App Icons
        const dock = document.getElementById('app-dock');
        const icons = [
            'fa-house', 'fa-gear', 'fa-heart-pulse', 'fa-cloud', 'fa-message', 
            'fa-camera', 'fa-map', 'fa-music', 'fa-phone', 'fa-wallet',
            'fa-calendar', 'fa-battery-full', 'fa-bluetooth', 'fa-wifi', 'fa-stopwatch',
            'fa-compass', 'fa-calculator', 'fa-envelope', 'fa-video', 'fa-star'
        ];
        
        icons.forEach((icon, index) => {
            const app = document.createElement('div');
            app.className = 'app-icon';
            app.innerHTML = `<i class="fa-solid ${icon}"></i>`;
            app.title = `App ${index + 1}`;
            dock.appendChild(app);
        });

        // Clock & Barometer Logic
        function updateDisplay() {
            const now = new Date();
            const h = now.getHours();
            const m = now.getMinutes();
            const s = now.getSeconds();
            const ms = now.getMilliseconds();

            // Analog Watch 1
            const sDeg = (s * 6) + (ms * 0.006);
            const mDeg = (m * 6) + (s * 0.1);
            const hDeg = ((h % 12) * 30) + (m * 0.5);

            document.getElementById('s1').style.transform = `rotate(${sDeg}deg)`;
            document.getElementById('m1').style.transform = `rotate(${mDeg}deg)`;
            document.getElementById('h1').style.transform = `rotate(${hDeg}deg)`;

            // Digital Watch 2
            const timeStr = `${h.toString().padStart(2, '0')}:${m.toString().padStart(2, '0')}`;
            document.getElementById('digi-time').innerText = timeStr;
            const dashOffset = 628 - (628 * (s / 60));
            document.getElementById('digi-progress').style.strokeDashoffset = dashOffset;

            // Minimalist Watch 3
            document.getElementById('s3').style.transform = `rotate(${sDeg}deg)`;
            document.getElementById('m3').style.transform = `rotate(${mDeg}deg)`;

            // Barometer jitter simulation
            const baseBaro = 1013;
            const jitter = Math.sin(Date.now() / 2000) * 2;
            const baroFinal = (baseBaro + jitter).toFixed(1);
            document.getElementById('baro-val').innerText = baroFinal;
            document.getElementById('baro-needle').style.transform = `rotate(${(jitter * 10) - 45}deg)`;

            requestAnimationFrame(updateDisplay);
        }

        updateDisplay();
    </script>
